// hooks/useCounter.js import { useState, useEffect, useRef } from 'react'; import AsyncStorage from '@react-native-async-storage/async-storage'; const STORAGE_KEY = '@counter_value'; export function useCounter(initialValue = 0) { const [count, setCount] = useState(initialValue); const [isAuto, setIsAuto] = useState(false); const intervalRef = useRef(null); // Відновлення значення після перезапуску useEffect(() => { const load = async () => { const saved = await AsyncStorage.getItem(STORAGE_KEY); if (saved !== null) setCount(Number(saved)); }; load(); }, []); // Збереження при кожній зміні useEffect(() => { AsyncStorage.setItem(STORAGE_KEY, String(count)); }, [count]); // Автоматичний режим: setInterval + clearInterval при cleanup useEffect(() => { if (isAuto) { intervalRef.current = setInterval(() => { setCount(prev => prev + 1); // функційне оновлення — безпечно }, 1000); } else { clearInterval(intervalRef.current); } // Cleanup при розмонтуванні або зміні isAuto return () => clearInterval(intervalRef.current); }, [isAuto]); return { count, isAuto, increment: () => setCount(c => c + 1), decrement: () => setCount(c => c - 1), incrementBy: (n) => setCount(c => c + n), decrementBy: (n) => setCount(c => c - n), reset: () => setCount(0), toggleAuto: () => setIsAuto(a => !a), }; }

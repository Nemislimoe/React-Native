// ContactForm.js export default function ContactForm({ onSave, onCancel, initial }) { const [name, setName] = useState(initial?.name ?? ''); const [phone, setPhone] = useState(initial?.phone ?? ''); const [email, setEmail] = useState(initial?.email ?? ''); const [errors, setErrors] = useState({}); const validate = () => { const e = {}; if (!name.trim()) e.name = 'Введіть імʼя'; if (!/^\+?\d{10,13}$/.test(phone)) e.phone = 'Невірний формат телефону'; if (!/\S+@\S+\.\S+/.test(email)) e.email = 'Невірний формат email'; setErrors(e); return Object.keys(e).length === 0; }; const handleSave = () => { if (!validate()) return; onSave({ name, phone, email }); }; return ( <View style={styles.form}> <TextInput value={name} onChangeText={setName} placeholder="Імʼя *" style={[styles.input, errors.name && styles.inputErr]} /> {errors.name && <Text style={styles.err}>{errors.name}</Text>} <TextInput value={phone} onChangeText={setPhone} placeholder="Телефон * (+380...)" keyboardType="phone-pad" style={[styles.input, errors.phone && styles.inputErr]} /> {errors.phone && <Text style={styles.err}>{errors.phone}</Text>} <TextInput value={email} onChangeText={setEmail} placeholder="Email *" keyboardType="email-address" style={[styles.input, errors.email && styles.inputErr]} /> {errors.email && <Text style={styles.err}>{errors.email}</Text>} <View style={styles.row}> <TouchableOpacity style={styles.btnPrimary} onPress={handleSave}> <Text style={styles.btnPrimaryText}>Зберегти</Text> </TouchableOpacity> <TouchableOpacity style={styles.btnSecondary} onPress={onCancel}> <Text>Скасувати</Text> </TouchableOpacity> </View> </View> ); } // ContactItem.js export default function ContactItem({ contact, onEdit, onDelete }) { return ( <View style={styles.row}> <View style={styles.avatar}> <Text style={styles.avatarText}> {contact.name[0].toUpperCase()} </Text> </View> <View style={styles.info}> <Text style={styles.name}>{contact.name}</Text> <Text style={styles.phone}>{contact.phone}</Text> <Text style={styles.email}>{contact.email}</Text> </View> <TouchableOpacity onPress={() => onEdit(contact)}> <Text>✏️</Text> </TouchableOpacity> <TouchableOpacity onPress={() => onDelete(contact.id)}> <Text>🗑️</Text> </TouchableOpacity> </View> ); } // App.js export default function App() { const [contacts, setContacts] = useState([]); const [editing, setEditing] = useState(null); const [showForm, setShowForm] = useState(false); const handleSave = (data) => { if (editing) { setContacts(contacts.map((c) => c.id === editing.id ? { ...c, ...data } : c )); } else { setContacts([...contacts, { ...data, id: Date.now() }]); } setShowForm(false); setEditing(null); }; const handleDelete = (id) => setContacts(contacts.filter((c) => c.id !== id)); const handleEdit = (contact) => { setEditing(contact); setShowForm(true); }; return ( <View> <FlatList data={contacts} keyExtractor={(c) => String(c.id)} renderItem={({ item }) => ( <ContactItem contact={item} onEdit={handleEdit} onDelete={handleDelete} /> )} /> {showForm ? <ContactForm initial={editing} onSave={handleSave} onCancel={() => setShowForm(false)} /> : <TouchableOpacity onPress={() => setShowForm(true)}> <Text>＋ Додати контакт</Text> </TouchableOpacity>} </View> ); }

// StarPicker.js — вибір зірок export default function StarPicker({ value, onChange }) { return ( <View style={styles.row}> {[1,2,3,4,5].map((n) => ( <TouchableOpacity key={n} onPress={() => onChange(n)}> <Text style={styles.star}> {n <= value ? '★' : '☆'} </Text> </TouchableOpacity> ))} </View> ); } // ReviewCard.js — один відгук export default function ReviewCard({ review, onEdit, onDelete }) { const [editing, setEditing] = useState(false); const [text, setText] = useState(review.text); const [rating, setRating] = useState(review.rating); if (editing) { return ( <View style={styles.card}> <StarPicker value={rating} onChange={setRating} /> <TextInput value={text} onChangeText={setText} style={styles.input} multiline /> <TouchableOpacity onPress={() => { onEdit({ ...review, text, rating }); setEditing(false); }}> <Text>Зберегти</Text> </TouchableOpacity> </View> ); } return ( <View style={styles.card}> <Text style={styles.stars}> {'★'.repeat(review.rating)} {'☆'.repeat(5 - review.rating)} </Text> <Text style={styles.text}>{review.text}</Text> <Text style={styles.date}>{review.date}</Text> <View style={styles.actions}> <TouchableOpacity onPress={() => setEditing(true)}> <Text>✏️ Редагувати</Text> </TouchableOpacity> <TouchableOpacity onPress={() => onDelete(review.id)}> <Text>🗑️ Видалити</Text> </TouchableOpacity> </View> </View> ); } // App.js export default function App() { const [reviews, setReviews] = useState([]); const [rating, setRating] = useState(0); const [text, setText] = useState(''); // Середній рейтинг const avg = reviews.length ? (reviews.reduce((s, r) => s + r.rating, 0) / reviews.length).toFixed(1) : null; const addReview = () => { if (!rating || !text.trim()) return; setReviews([{ id: Date.now(), rating, text, date: new Date().toLocaleDateString('uk-UA'), }, ...reviews]); setRating(0); setText(''); }; const editReview = (updated) => setReviews(reviews.map((r) => r.id === updated.id ? updated : r)); const deleteReview = (id) => setReviews(reviews.filter((r) => r.id !== id)); return ( <ScrollView> <Text>Середній рейтинг: {avg ?? '—'}</Text> <StarPicker value={rating} onChange={setRating} /> <TextInput value={text} onChangeText={setText} placeholder="Ваш відгук..." multiline /> <Button title="Додати відгук" onPress={addReview} /> {reviews.map((r) => ( <ReviewCard key={r.id} review={r} onEdit={editReview} onDelete={deleteReview} /> ))} </ScrollView> ); }

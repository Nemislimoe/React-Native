// DynamicHeading.js — дочірній компонент import React from 'react'; import { Text, StyleSheet } from 'react-native'; export default function DynamicHeading({ title, color }) { return ( <Text style={[styles.heading, { color }]}> {title} </Text> ); } const styles = StyleSheet.create({ heading: { fontSize: 28, fontWeight: 'bold', textAlign: 'center', marginVertical: 16, }, }); // ───────────────────────────────────────── // App.js — батьківський компонент import React, { useState } from 'react'; import { View, TextInput, StyleSheet } from 'react-native'; import DynamicHeading from './DynamicHeading'; const COLORS = [ '#6C63FF', '#1D9E75', '#D85A30', '#185FA5' ]; export default function App() { const [title, setTitle] = useState('Привіт, Світ!'); const [color, setColor] = useState(COLORS[0]); return ( <View style={styles.container}> {/* Передаємо пропси в дочірній компонент */} <DynamicHeading title={title} color={color} /> <TextInput style={styles.input} value={title} onChangeText={setTitle} placeholder="Введіть заголовок" /> <View style={styles.colorRow}> {COLORS.map((c) => ( <TouchableOpacity key={c} style={[styles.colorBtn, { backgroundColor: c }]} onPress={() => setColor(c)} /> ))} </View> </View> ); }

// TaskItem.js — компонент одного завдання import React from 'react'; import { TouchableOpacity, Text, View, StyleSheet } from 'react-native'; export default function TaskItem({ task, onToggle }) { return ( <TouchableOpacity style={[ styles.item, task.done && styles.itemDone ]} onPress={() => onToggle(task.id)} > <View style={[ styles.checkbox, task.done && styles.checkboxDone ]}> {task.done && <Text style={styles.check}>✓</Text>} </View> <Text style={[ styles.text, task.done && styles.textDone ]}> {task.text} </Text> </TouchableOpacity> ); } // ───────────────────────────────────────── // TaskList.js — компонент списку import React from 'react'; import { View, FlatList } from 'react-native'; import TaskItem from './TaskItem'; export default function TaskList({ tasks, onToggle }) { return ( <FlatList data={tasks} keyExtractor={(item) => String(item.id)} renderItem={({ item }) => ( <TaskItem task={item} onToggle={onToggle} /> )} /> ); } // ───────────────────────────────────────── // App.js export default function App() { const [tasks, setTasks] = useState([ { id: 1, text: 'Вивчити React Native', done: false }, { id: 2, text: 'Зрозуміти JSX', done: false }, { id: 3, text: 'Зробити домашнє завдання', done: false }, ]); const toggleTask = (id) => { setTasks(tasks.map((t) => t.id === id ? { ...t, done: !t.done } : t )); }; return ( <View style={styles.container}> <TaskList tasks={tasks} onToggle={toggleTask} /> </View> ); }

// ThemedCard.js — дочірній компонент import React from 'react'; import { View, Text, StyleSheet } from 'react-native'; export default function ThemedCard({ bgColor, textColor }) { return ( <View style={[styles.card, { backgroundColor: bgColor }]}> <Text style={[styles.title, { color: textColor }]}> Поточна тема </Text> <Text style={[styles.body, { color: textColor }]}> Це дочірній компонент, який отримує колір через пропси. </Text> </View> ); } // ───────────────────────────────────────── // App.js — батьківський компонент import React, { useState } from 'react'; import { View, TouchableOpacity, Text } from 'react-native'; import ThemedCard from './ThemedCard'; const THEMES = { light: { bg: '#FFFFFF', cardBg: '#F4F4F4', text: '#111111', btnBg: '#222222', btnText: '#FFFFFF', label: '🌙 Темна тема', }, dark: { bg: '#1A1A2E', cardBg: '#16213E', text: '#EAEAEA', btnBg: '#E2E2E2', btnText: '#111111', label: '☀️ Світла тема', }, }; export default function App() { const [isDark, setIsDark] = useState(false); const theme = isDark ? THEMES.dark : THEMES.light; return ( <View style={[styles.container, { backgroundColor: theme.bg }]}> {/* Передаємо кольори через пропси */} <ThemedCard bgColor={theme.cardBg} textColor={theme.text} /> <TouchableOpacity style={[styles.btn, { backgroundColor: theme.btnBg }]} onPress={() => setIsDark(!isDark)} > <Text style={{ color: theme.btnText }}> {theme.label} </Text> </TouchableOpacity> </View> ); }

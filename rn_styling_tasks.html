// utils/validation.js export function validate(form) { const errors = {}; // Валідація імені if (!form.name.trim()) { errors.name = "Поле обов'язкове"; } // Валідація email через регулярний вираз const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!emailRegex.test(form.email)) { errors.email = 'Невірний формат email'; } // Валідація пароля if (form.password.length < 6) { errors.password = 'Мінімум 6 символів'; } // Збіг паролів if (form.password !== form.confirm) { errors.confirm = 'Паролі не збігаються'; } return errors; }
